              DHT11 dht11(2);

              const int ANALOG_OUTPUT = 11;

              void setup() {
                  // Initialize serial communication to allow debugging and data readout.
                  // Using a baud rate of 9600 bps.
                  Serial.begin(9600);

              void loop() {
                  int temperature = 0;
                  int humidity = 0;

                  // Attempt to read the temperature and humidity values from the DHT11 sensor.
                  int result = dht11.readTemperatureHumidity(temperature, humidity);

                  // Check the results of the readings.
                  // If the reading is successful, print the temperature and humidity values.
                  // If there are errors, print the appropriate error messages.
                  if (result == 0) {

                      int output1 = map(temperature, 0, 100, 0, 255); // Convert sensor and humdiity to analog scale
                      int output2 = map(humidity, 0, 100, 0, 255);
                      Serial.print("Temperature: ");
                      Serial.print(temperature); // Print info
                      int analogOutput = (output1 + output2) / 2; // Print and log info
                      analogWrite(ANALOG_OUTPUT, analogOutput); // WRite to the Arudino board so that it controls current in LED
                      Serial.print(" Â°C\tHumidity: ");

                  } else {
                      // Print error message based on the error code.

              }
